# Description: Create the PUT endpoint to update existing deals in the pipeline.
# Details:
Implement the PUT function in the route.ts file that:
1. Creates a Supabase client using environment variables
2. Extracts deal ID and update data from the request body
3. Updates the specified deal in the pipeline_deals table
4. Returns the updated deal or an error response

```typescript
export async function PUT(request: Request) {
  const supabase = createClient(
    process.env.SUPABASE_URL!,
    process.env.SUPABASE_KEY!
  );
  const { id, ...updateData } = await request.json();

  const { data, error } = await supabase
    .from('pipeline_deals')
    .update(updateData)
    .eq('id', id)
    .select();

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  return NextResponse.json(data[0]);
}
```
