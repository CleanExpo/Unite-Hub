```typescript
// Error handling utility
function handleError(error: any) {
  console.error('API Error:', error);
  
  // Determine if this is a client error or server error
  const isClientError = error.code === 'PGRST116' || error.message?.includes('validation');
  const status = isClientError ? 400 : 500;
  
  // Create safe error message
  const safeMessage = isClientError 
    ? error.message 
    : 'An unexpected error occurred';
  
  return NextResponse.json({ error: safeMessage }, { status });
}

// Example usage in an endpoint
export async function GET() {
  const supabase = createClient(
    process.env.SUPABASE_URL!,
    process.env.SUPABASE_KEY!
  );

  try {
    const { data: deals, error } = await supabase
      .from('pipeline_deals')
      .select('*')
      .order('created_at', { ascending: false });

    if (error) throw error;
    
    console.log(`Retrieved ${deals.length} deals`);
    return NextResponse.json(deals);
  } catch (error) {
    return handleError(error);
  }
}
```
